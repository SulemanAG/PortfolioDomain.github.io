Description

This C++ program implements various algorithms for smart city management. It provides solutions for different aspects of urban planning and optimization, including:

  • Optimal Tourist Routes using graph algorithms
  • Menu Item Sorting using Quick Sort
  • Traffic Monitoring using Breadth-First Search (BFS)
  • Waste Management Route Optimization using Dijkstra's algorithm
  • Water Usage Analysis using Time Series
  • Internet Connectivity Optimization using Minimum Spanning Tree (MST)

The program features a user-friendly menu interface, allowing users to select and run specific algorithms based on their needs. Each function takes user input to create and manipulate data structures relevant to its purpose, demonstrating practical applications of these algorithms in a smart city context.

// Function to optimize waste management routes using Dijkstra's Algorithm
void optimizeWasteManagementRoutes() {
    int n, e;
    cout << "Enter the number of locations (nodes) and routes (edges): ";
    cin >> n >> e;

    vector>> graph(n);
    cout << "Enter the routes in the format: source destination weight (e.g., 0 1 4):\n";
    for (int i = 0; i < e; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].emplace_back(v, w);
        graph[v].emplace_back(u, w); // Assuming undirected graph
    }

    vector dist(n, INT_MAX);
    dist[0] = 0;
    priority_queue, vector>, greater<>> pq;
    pq.push({0, 0});

    while (!pq.empty()) {
        auto [cost, node] = pq.top();
        pq.pop();
        for (auto [next, weight] : graph[node]) {
            if (dist[next] > cost + weight) {
                dist[next] = cost + weight;
                pq.push({dist[next], next});
            }
        }
    }

    cout << "Shortest distances from the central hub (node 0):\n";
    for (int i = 0; i < n; ++i) {
        cout << "Location " << i << ": " << dist[i] << "\n";
    }
}

int main() {
    optimizeWasteManagementRoutes();
    return 0;
}

// Function to optimize internet connectivity using Minimum Spanning Tree (MST)
void optimizeCityInternetConnectivity() {
    int n;
    cout << "Enter the number of locations (nodes): ";
    cin >> n;

    vector> graph(n, vector(n, 0));
    cout << "Enter the adjacency matrix (cost of connecting nodes, use 0 if not connected):\n";
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> graph[i][j];
        }
    }

    vector key(n, INT_MAX);
    vector inMST(n, false);
    key[0] = 0;
    int result = 0;

    for (int count = 0; count < n - 1; ++count) {
        int u = -1;
        for (int v = 0; v < n; ++v) {
            if (!inMST[v] && (u == -1 || key[v] < key[u])) {
                u = v;
            }
        }

        inMST[u] = true;
        result += key[u];

        for (int v = 0; v < n; ++v) {
            if (graph[u][v] && !inMST[v] && graph[u][v] < key[v]) {
                key[v] = graph[u][v];
            }
        }
    }

    cout << "Total cost for optimized connectivity: " << result << "\n";
}

int main() {
    optimizeCityInternetConnectivity();
    return 0;
}
